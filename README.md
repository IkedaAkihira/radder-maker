# タイムチャートのJSON
```json
{
	"abs":true,             //タイムチャートの値が絶対的なものか相対的なものかどうか。
	"SW1":[1,2,5,6],
	"SW2":[3,4,7,8],
	"L1":[
        {
            "ref":"SW1",
            "index":0
        },                  //オブジェクトを用いて、他の値を参照することが可能。これによって、保守性が上がる。
        {
            "ref":"SW2",
            "index":0
        },
        {
            "ref":"SW1",
            "index":2
        },
        {
            "ref":"SW2",
            "index":2
        }
        ],
    "L2":[
        {
            "ref":"SW2",
            "index":0
        },
        {
            "ref":"SW1",
            "index":2
        },
        {
            "ref":"SW2",
            "index":2
        }
    ]
}
```
